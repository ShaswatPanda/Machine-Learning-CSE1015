    def encode_target(df, target_column):
        df_mod = df.copy()
        targets = df_mod[target_column].unique()
        map_to_int = {name: n for n, name in enumerate(targets)}
        df_mod["genre.encode"] = df_mod[target_column].replace(map_to_int)
        return (df_mod, targets)
